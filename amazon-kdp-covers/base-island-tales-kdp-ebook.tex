\newcommand{\drawcover}{%
\pagecolor{backgroundcolor}

\begin{tikzpicture}[remember picture,overlay]
    % Page coordinates
    \coordinate (page center) at (current page.center);
    \coordinate (page north) at (current page.north);
    \coordinate (page south) at (current page.south);
    \coordinate (page west) at (current page.west);
    \coordinate (page east) at (current page.east);
    
    % Central bar now rooted at left edge
    \coordinate (bar left) at (page west);
    \coordinate (bar right) at ([xshift=\centralbarwidth]page west);
    
    % Draw vertical bar at left edge
    \fill[centralbarcolor] (bar left |- page north) rectangle (bar right |- page south);
    
    % Right side text centered between bar and right edge
    \node[anchor=center, text=righttextcolor, align=left, text width=\rightsidewidth] 
        at ($ (bar right) !0.5! (page east) + (0+1.0cm, \rightsidetop) $) {
        \righttextfont
        \baselineskip=2em 
        \rightsidetext
    };
    
    % Right spine text (rotated)
    \node[anchor=center, text=rightspinecolor, rotate=-90] 
        at ($ (bar left) !0.5! (bar right) $) {
        \rightspinefont
        \rightspinetext
    };
    

\node[anchor=center, text=footertextcolor, align=center, text width=\footerwidth] 
    at ($ (bar right) !0.5! (page east) + (0, \footerupdist) $) {
    \footerfont
    \baselineskip=1.5em
    \bookcoverfooter
};

\end{tikzpicture}
}
